from sklearn.preprocessing import StandardScaler
from scipy.spatial.distance import euclidean

# Select trial stores
trial_stores = [77, 86, 88]

# Define weights for 'MONTHLY_OVERALL_SALES' and 'MONTHLY_NO._OF_CUSTOMERS'
sales_weight = 0.5
customers_weight = 0.5

# Calculate combined score
merged_df['COMBINED_SCORE'] = (sales_weight * merged_df['MONTHLY_OVERALL_SALES'] + 
                               customers_weight * merged_df['MONTHLY_NO._OF_CUSTOMERS'])

# Normalize the data
scaler = StandardScaler()
merged_df[['MONTHLY_OVERALL_SALES', 'MONTHLY_NO._OF_CUSTOMERS', 'COMBINED_SCORE']] = scaler.fit_transform(
    merged_df[['MONTHLY_OVERALL_SALES', 'MONTHLY_NO._OF_CUSTOMERS', 'COMBINED_SCORE']])

# Find control stores for each trial store
control_stores = {}
for trial_store in trial_stores:
    trial_store_data = merged_df[merged_df['STORE_NBR'] == trial_store].iloc[0]
    trial_store_vector = trial_store_data[['MONTHLY_OVERALL_SALES', 'MONTHLY_NO._OF_CUSTOMERS', 'COMBINED_SCORE']]
    
    # Calculate similarity scores
    merged_df['SIMILARITY_SCORE'] = merged_df.apply(
        lambda x: euclidean(trial_store_vector, x[['MONTHLY_OVERALL_SALES', 'MONTHLY_NO._OF_CUSTOMERS', 'COMBINED_SCORE']]), 
        axis=1)
    
    # Exclude trial store itself from the list of control stores
    similar_stores = merged_df[merged_df['STORE_NBR'] != trial_store]
    
    # Select top 3 control stores
    control_stores[trial_store] = similar_stores.sort_values('SIMILARITY_SCORE')['STORE_NBR'].head(3).tolist()

print("Control stores for each trial store:")
for trial_store, control_store_list in control_stores.items():
    print(f"Trial Store {trial_store}: {control_store_list}")

# Calculate similarity between trial stores and potential control stores
trial_control_similarity = {}
for trial_store, control_store_list in control_stores.items():
    trial_store_data = merged_df[merged_df['STORE_NBR'] == trial_store][['MONTHLY_OVERALL_SALES', 'MONTHLY_NO._OF_CUSTOMERS']]
    control_store_data = merged_df[merged_df['STORE_NBR'].isin(control_store_list)][['MONTHLY_OVERALL_SALES', 'MONTHLY_NO._OF_CUSTOMERS']]
    
    # Check for any potential issues in the data
    if trial_store_data.empty or control_store_data.empty:
        print(f"Trial Store {trial_store} or its control stores have no data.")
        continue
    
    # Calculate cosine similarity
    try:
        similarity_matrix = cosine_similarity(trial_store_data, control_store_data)

        print(f"Error calculating similarity for Trial Store {trial_store} and its control stores: {e}")
        continue

import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

# Trial and control store mappings
trial_control_stores = {
    77: 233,
    86: 55,
    88: 26
}

# Step 1: Define a function to generate the plot for each pair of trial and control stores
def plot_trial_control_sales(trial_store, control_store):
    # Filter the DataFrame for the trial and control stores
    filtered_df = merged_df[(merged_df['STORE_NBR'] == trial_store) | (merged_df['STORE_NBR'] == control_store)]
    
    # Group by month and calculate total sales
    monthly_sales = filtered_df.groupby(['MONTH', 'STORE_NBR'])['MONTHLY_OVERALL_SALES'].sum().unstack()
    
    # Convert month name to numerical representation
    month_order = ['July', 'August', 'September', 'October', 'November', 'December', 'January', 'February', 'March', 'April', 'May', 'June']
    monthly_sales.index = pd.CategoricalIndex(monthly_sales.index, categories=month_order, ordered=True)
    monthly_sales = monthly_sales.sort_index()
    
    # Plot the total sales
    plt.figure(figsize=(12, 6))
    
    # Plotting side by side bar plots for each store
    monthly_sales.plot(kind='bar', color=['skyblue', 'orange'], width=0.4, position=0.5)
    
    # Adding trendlines for each store's sales
    for store_nbr in monthly_sales.columns:
        z = np.polyfit(range(len(monthly_sales)), monthly_sales[store_nbr], 1)
        p = np.poly1d(z)
        if store_nbr == control_store:
            plt.plot(range(len(monthly_sales)), p(range(len(monthly_sales))), linestyle='-.', color='red')  # Different linestyle for control store
        else:
            plt.plot(range(len(monthly_sales)), p(range(len(monthly_sales))), linestyle='--', color='grey')  # Default linestyle
    
    plt.title(f'Monthly Overall Sales of STORE_NBR {trial_store} and {control_store} (July-June)')
    plt.xlabel('Month')
    plt.ylabel('Monthly Overall Sales')
    plt.xticks(range(len(monthly_sales)), month_order, rotation=45)
    plt.legend(title='Store', labels=[f'Trial Store {trial_store}', f'Control Store {control_store}'], loc='upper left')
    plt.grid(True, linestyle='--', alpha=0.7)
    plt.tight_layout()
    plt.show()

# Step 2: Loop through each trial and control store pair and generate the plot
for trial_store, control_store in trial_control_stores.items():
    plot_trial_control_sales(trial_store, control_store)

    # Filter the DataFrame for the trial and control stores
    trial_df = merged_df[merged_df['STORE_NBR'] == trial_store]['MONTHLY_OVERALL_SALES']
    control_df = merged_df[merged_df['STORE_NBR'] == control_store]['MONTHLY_OVERALL_SALES']
    
    # Perform t-test
    t_statistic, p_value = ttest_rel(trial_df, control_df)
    
    # Print results
    print(f"Trial Store {trial_store} vs Control Store {control_store}:")
    print(f"    t-statistic: {t_statistic}")
    print(f"    p-value: {p_value}")
    
    # Compare p-value to alpha
    if p_value < alpha:
        print("    Conclusion: Reject null hypothesis. Trial store acts significantly different from its control store.")
    else:
        print("    Conclusion: Fail to reject null hypothesis. No significant difference observed between trial and control store.")
